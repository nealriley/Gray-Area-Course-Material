An example of a program that uses the arrow keys


Press the arrow keys to change the color of the circle
You may need to click on the applet to give it the focus 



void setup()
{
}
void draw()
{
  ellipse(50,50,40,40);
}
void keyPressed()
{
  if(keyCode == UP)
  {
    fill(255,0,0);
  }
  else if(keyCode == DOWN)
  {
    fill(255,255,0);
  }
  else if(keyCode == LEFT)
  {
    fill(0,255,255);
  }
  else if(keyCode == RIGHT)
  {
    fill(0,0,255);
  }
}
